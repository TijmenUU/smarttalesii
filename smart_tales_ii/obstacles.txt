# <-- This indicates a comment line, it is ignored by the game
# This file defines the obstacle types and their properties
# The game uses a 1280x720 fixed internal resolution, independent
# of the actual window size

# The following per line layout is used:
# obstacle_type obstacle_texture_file_location interaction_radius minimal_distance swipe_gesture ...
# A line may have multiple swipe gestures at the end

# Obstacle Types
# The following obstacle types are supported:
# - furniture
# - door
# - lightswitch
# - phone

# interaction_radius
# Unit: pixels
# How nearby does the gesture have to start in order to be valid?

# minimal_distance
# Unit: pixels
# How much distance should there be between the start and end point
# of a gesture? This is to distinguish taps from swipes. If set really
# high it will almost always result in a tap, if set really low it will 
# almost never be a tap.

# Gestures
# The following gestures are supported:
# - tap
# - upwards
# - downwards
# - leftwards
# - rightwards

# DEFINITIONS
furniture	texture/furniture.png		200	50	downwards
door		texture/door.png			200	50	leftwards rightwards
lightswitch	texture/lightswitch.png	100	100	tap
phone		texture/phone.png			200	50	upwards
