# Animation properties available:
# frame-start-x
#	Unit: pixels
# frame-start-y
#	Unit: pixels
# frame-width
#	Unit: pixels
# frame-height
#	Unit: pixels
# frame-margin
#	Unit: pixels
#	Added when getting the next frame, fixes potential artifacting with transparancy
# frame-count
# frame-time
#	in milliseconds
# reverse
#	If present, it sets the animation to reverse play (start position - (margin + frame width))

	frame-start-y 0
	frame-width 16
